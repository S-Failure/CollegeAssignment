package com.example.ass3;

import androidx.appcompat.app.AppCompatActivity;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.EditText;
import android.widget.Spinner;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.RadioButton;
import android.widget.RadioGroup;

public class MainActivity extends AppCompatActivity {

    Spinner sp;
    EditText edt1,edt2,edt3;

    String[] batch = {"2016-2017","2017-2018","2018-2019","2019-2020"};
    String result;

    EditText editTextEmail, editTextPass, editTextMobile, editTextName;
    RadioButton radioButtonMale, radioButtonFemale;
    RadioGroup radioGroupGender;
    Spinner spinnerBatch;
    CheckBox chkWeb, chkSoftware, chkAI, chkML;
    Button buttonSubmit;
    String mEmail, mPass, mMobile, mName, mGender, mBatch, mChoice;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        editTextEmail = findViewById(R.id.Email);
        editTextMobile = findViewById(R.id.Phone);
        editTextName = findViewById(R.id.FullName);
        editTextPass = findViewById(R.id.Password);
        radioGroupGender = findViewById(R.id.Gender);
        radioButtonFemale = findViewById(R.id.Female);
        radioButtonMale = findViewById(R.id.Male);
        spinnerBatch = findViewById(R.id.Spinner);
        final String[] list = {"2010", "2011", "2012", "2013", "2014", "2015", "2016", "2017", "2018", "2019", "2020"};
        ArrayAdapter<String> adapter = new ArrayAdapter(this, R.layout.support_simple_spinner_dropdown_item, list);
        spinnerBatch.setAdapter(adapter);
        chkWeb = findViewById(R.id.chkWeb);
        chkSoftware = findViewById(R.id.chkSoftware);
        chkAI = findViewById(R.id.chkAI);
        chkML = findViewById(R.id.chkML);
        buttonSubmit = findViewById(R.id.Submit);
        spinnerBatch.setOnItemSelectedListener(new AdapterView.OnItemSelectedListener() {
            @Override
            public void onItemSelected(AdapterView<?> parent, View view, int position, long id) {
                mBatch = list[position].toString();
            }

            @Override
            public void onNothingSelected(AdapterView<?> parent) {

            }
        });

    }

    public void Submit(View view)
    {
        mEmail = editTextEmail.getText().toString();
        mMobile = editTextMobile.getText().toString();
        mName = editTextName.getText().toString();
        final String mGender =
                ((RadioButton)findViewById(radioGroupGender.getCheckedRadioButtonId()))
                        .getText().toString();
        if (chkWeb.isChecked()) {
            mChoice = chkWeb.getText().toString();
        }
        if (chkSoftware.isChecked()) {
            mChoice += chkSoftware.getText().toString();
        }
        if (chkAI.isChecked()) {
            mChoice += chkAI.getText().toString();
        }
        if (chkML.isChecked()) {
            mChoice += chkML.getText().toString();
        }

        Intent intent = new Intent(this, ShowData.class);
        intent.putExtra("Name", mName);
        intent.putExtra("Email", mEmail);

        intent.putExtra("Choice", mChoice);
        intent.putExtra("Batch", mBatch);
        intent.putExtra("Gender", mGender);
        intent.putExtra("Mobile", mMobile);
        startActivity(intent);
    }

}

package com.example.ass3;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.widget.TextView;
import android.content.Intent;

public class ShowData extends AppCompatActivity {

    TextView textViewName, textViewEmail, textViewPass, textViewMobile, textViewBatch, textViewCourse, textViewGender;
    String name,email,pass,mobile,batch,course,gender;


    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_show_data);

        textViewBatch = findViewById(R.id.batch);
        textViewCourse = findViewById(R.id.course);
        textViewEmail = findViewById(R.id.email);
        textViewGender = findViewById(R.id.gender);
        textViewName = findViewById(R.id.name);
        textViewPass = findViewById(R.id.pass);
        textViewMobile = findViewById(R.id.mobile);
        Intent intent = getIntent();
        name = intent.getStringExtra("Name");
        email = intent.getStringExtra("Email");
        pass = intent.getStringExtra("Pass");
        course = intent.getStringExtra("Choice");
        batch = intent.getStringExtra("Batch");
        gender = intent.getStringExtra("Gender");
        mobile = intent.getStringExtra("Mobile");

        textViewMobile.setText(mobile);
        textViewCourse.setText(course);
        textViewPass.setText(pass);
        textViewName.setText(name);
        textViewGender.setText(gender);
        textViewEmail.setText(email);
        textViewBatch.setText(batch);
    }
}
